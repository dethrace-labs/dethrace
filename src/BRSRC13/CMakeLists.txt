    PUBLIC

        -Wall
        -Werror

        -Wno-unused-variable
        -Wno-unused-parameter
        -Wno-sign-compare
        -Wno-int-conversion  #todo!
        -Wno-strict-aliasing
    )
    
    target_link_libraries(brender PUBLIC m)
else()
    target_compile_definitions(brender PRIVATE -D_CRT_SECURE_NO_WARNINGS)
    target_compile_options(brender PRIVATE
        /wd4101
        /wd4996
    )
endif()
